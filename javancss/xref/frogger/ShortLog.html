26  package frogger;
27  import jig.engine.util.Vector2D;
28    
29  public class ShortLog extends MovingEntity {
30  	
31  	public static int LENGTH = 32*3;
32  	
33  	public ShortLog (Vector2D pos, Vector2D v) {
34  		super(Main.SPRITE_SHEET + "#shortlog");
35  		position = pos;
36  		Vector2D posSphere1 = position;
37  		Vector2D posSphere2 = new Vector2D(position.getX()+32, position.getY());
38  		Vector2D posSphere3 = new Vector2D(position.getX()+64, position.getY());
39  		collisionObjects.add(new CollisionObject("colSmall", posSphere1));
40  		collisionObjects.add(new CollisionObject("colSmall", posSphere2));
41  		collisionObjects.add(new CollisionObject("colSmall", posSphere3));
42  		velocity = v;
43  		if (v.getX() < 0)
44  			setFrame(1);
45  		else
46  			setFrame(0);
47  	}
48  
49  }
