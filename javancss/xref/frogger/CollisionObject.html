26  package frogger;
27  import jig.engine.physics.vpe.VanillaSphere;
28  import jig.engine.util.Vector2D;
29  
30  
31  public class CollisionObject extends VanillaSphere {
32  
33  	public CollisionObject(Vector2D pos) {
34  		super("col");
35  		setPosition(pos);
36  	}
37  	
38  	public CollisionObject(String name, Vector2D pos) {
39  		super(name);
40  		setPosition(pos);
41  	}
42  	
43  	/**
44  	 * Depending on the collision sphere, we offset it's position so
45  	 * that it appears in the middle of the object
46  	 */
47  	public void setPosition(Vector2D pos) {
48  		double dX = 16 - getRadius();
49  		double dY = -getRadius() +16;
50  		position = new Vector2D(pos.getX()+dX, pos.getY()+dY);
51  	}
52  	
53  	public void update(long deltaMs) {
54  		;
55  	}
56  }
