26  package frogger;
27  
28  import jig.engine.util.Vector2D;
29  
30  /**
31   * Simple tiny particle
32   * 
33   * Used to show some weather effects
34   * 
35   * @author vitaliy
36   *
37   */
38  public class Particle extends MovingEntity {
39  	
40  	private int timeExpire = 1;
41  	private int timeAlive = 1;
42  	
43  	public Particle(String sprite, Vector2D pos, Vector2D v) {
44  		super(sprite);
45  		position = pos;
46  		velocity = v;
47  		setActivation(true);
48  		timeExpire = 0;
49  	}
50  	
51  	/**
52  	 * Build particle with expiration timer
53  	 * 
54  	 * @param pos - position
55  	 * @param v - velocity
56  	 * @param te - expiration timer in milliseconds
57  	 */
58  	public Particle(String sprite, Vector2D pos, Vector2D v, int te) {
59  		super(sprite);
60  		position = pos;
61  		velocity = v;
62  		setActivation(true);
63  		timeExpire = te;
64  	}
65  	
66  	public void update(final long deltaMs) {
67  		super.update(deltaMs);
68  		
69  		// Check the expiration
70  		if (timeExpire != 0) {
71  			timeAlive += deltaMs;
72  			if (timeAlive > timeExpire)
73  				setActivation(false);
74  		}
75  	}
76  }
