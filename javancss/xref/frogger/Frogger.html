26  package frogger;
27  
28  import jig.engine.util.Vector2D;
29  
30  /**
31   * Main sprite in the game that a player can control
32   * 
33   * @author vitaliy
34   *
35   */
36  public class Frogger extends MovingEntity {
37  	
38  	final static int MOVE_STEP = 32;
39  	
40  	// Animation related variables 
41  	final static private int ANIMATION_STEP = 4; // 32/4 = 8, 8 animation frames, 10 ms each
42  	
43  	private int curAnimationFrame = 0;
44  	private int finalAnimationFrame = 0;
45  	private long animationDelay = 10; // milliseconds
46  	private long animationBeginTime = 0;
47  	private boolean isAnimating = false;
48  	private Vector2D dirAnimation = new Vector2D(0,0);
49  	
50  	// Object to follow, such as Tree Log in the river
51  	private MovingEntity followObject = null;
52  	
53  	
54  	public boolean isAlive = false;
55      private long timeOfDeath = 0;
56      
57      // Current sprite frame displayed
58      private int currentFrame = 0;
59      private int tmpFrame = 0;
60      
61      public int deltaTime = 0;
62      
63      public boolean cheating = false;
64      
65      public boolean hw_hasMoved = false;
66      
67      private Main game;
68      
69      /**
70       * Build frogger!
71       */
72  	public Frogger (Main g) {
73  		super(Main.SPRITE_SHEET + "#frog");
74  		game = g;
75  		resetFrog();
76  		collisionObjects.add(new CollisionObject(position));
77  	}
78  	
79  	/**
80  	 * Reset the Frogger to default state and position
81  	 */
82  	public void resetFrog() {
83  		isAlive = true;
84  		isAnimating = false;
85  		currentFrame = 0;
86  		followObject = null;
87  		position = Main.FROGGER_START;
88  		game.levelTimer = Main.DEFAULT_LEVEL_TIME;
89  	}
90  	
91  	/**
92  	 * Moving methods, called from Main upon key strokes
93  	 */
94  	public void moveLeft() {
95  		if (getCenterPosition().getX()-16 > 0 && isAlive && !isAnimating) {
96  			currentFrame = 3;
97  		    move(new Vector2D(-1,0));
98  		    AudioEfx.frogJump.play(0.2);
99  		}
100 	}
101 	
102 	public void moveRight() {
103 		
104 		if (getCenterPosition().getX()+32 < Main.WORLD_WIDTH && isAlive && !isAnimating) {
105 			currentFrame = 2;
106 		    move(new Vector2D(1,0));
107 		    AudioEfx.frogJump.play(0.2);
108 		}
109 	}
110 	
111 	public void moveUp() {
112 		if (position.getY() > 32  && isAlive && !isAnimating) {
113 			currentFrame = 0;
114 		    move(new Vector2D(0,-1));
115 		    AudioEfx.frogJump.play(0.2);
116 		}
117 	}
118 	
119 	public void moveDown() {
120 		if (position.getY() < Main.WORLD_HEIGHT - MOVE_STEP && isAlive && !isAnimating) {
121 			currentFrame = 1;
122 		    move(new Vector2D(0,1));
123 		    AudioEfx.frogJump.play(0.2);
124 		}
125 	}
126 	
127 	/**
128 	 * Short-cut for systems current time
129 	 * @return
130 	 */
131 	public long getTime() {
132 		return System.currentTimeMillis();
133 	}
134 	
135 	/**
136 	 * Initiate animation sequence into specified direction, given by
137 	 * 
138 	 * @param dir - specifies direction to move
139 	 * 
140 	 * The collision sphere of Frogger is automatically moved to the final
141 	 * position. The animation then lags behind by a few seconds(or frames). 
142 	 * This resolves the positioning bugs when objects collide during the animation.
143 	 */
144 	public void move(Vector2D dir) {
145 		followObject = null;
146 		curAnimationFrame = 0;
147 		finalAnimationFrame = MOVE_STEP/ANIMATION_STEP;
148 		isAnimating = true;
149 		hw_hasMoved = true;
150 		animationBeginTime = getTime();
151 		dirAnimation = dir;
152 
153 		tmpFrame = currentFrame;
154 		currentFrame += 5;
155 		
156 		// Move CollisionSphere to an already animated location
157 		sync(new Vector2D(
158 				position.getX()+dirAnimation.getX()*MOVE_STEP, 
159 				position.getY()+dirAnimation.getY()*MOVE_STEP)
160 		);
161 	}
162 	
163 	/**
164 	 * Cycle through the animation frames
165 	 */
166 	public void updateAnimation() {
167 		// If not animating, sync position of the sprite with its collision sphere
168 		if (!isAnimating || !isAlive) {
169 			sync(position);
170 			return;
171 		}
172 		
173 		// Finish animating
174 		if (curAnimationFrame >= finalAnimationFrame) {
175 			isAnimating = false;
176 			currentFrame = tmpFrame;
177 			return;
178 		}
179 		
180 		// Cycle animation
181 		if (animationBeginTime + animationDelay < getTime()) {
182 			animationBeginTime = getTime();
183 			position = new Vector2D(
184 					position.getX() + dirAnimation.getX()*ANIMATION_STEP,
185 					position.getY() + dirAnimation.getY()*ANIMATION_STEP
186 					);
187 			curAnimationFrame++;
188 			return;
189 		}
190 	}
191 	
192 	/**
193 	 * Re-align frog to a grid
194 	 */
195 	public void allignXPositionToGrid() {
196 		if (isAnimating || followObject != null) 
197 			return;
198 		double x = position.getX();
199 		x = Math.round(x/32)*32;
200 		position = new Vector2D(x, position.getY());
201 		
202 	}
203 	
204 	/**
205 	 * Following a Tree Log on a river by getting it's velocity vector
206 	 * @param deltaMs
207 	 */
208 	public void updateFollow(long deltaMs) {
209 		if (followObject == null || !isAlive) 
210 			return;
211 		Vector2D dS = followObject.getVelocity().scale(deltaMs);
212 		position = new Vector2D(position.getX()+dS.getX(), position.getY()+dS.getY());
213 	}
214 	
215 	/**
216 	 * Setting a moving entity to follow
217 	 * @param log
218 	 */
219 	public void follow(MovingEntity log) {
220 		followObject = log;
221 	}
222 	
223 	
224 	/**
225 	 * Effect of a wind gust on Frogger
226 	 * @param d
227 	 */
228 	public void windReposition(Vector2D d) {
229 		if (isAlive) {
230 			hw_hasMoved = true;
231 			setPosition(new Vector2D(getPosition().getX()+d.getX(), getPosition().getY()));
232 			sync(position);
233 		}
234 	}
235 	
236 	/**
237 	 * Effect of Heat Wave on Frogger
238 	 * @param randDuration
239 	 */
240 	public void randomJump(final int rDir) {
241 		switch(rDir) {
242 		case 0:
243 			moveLeft();
244 			break;
245 		case 1:
246 			moveRight();
247 			break;
248 		case 2:
249 			moveUp();
250 			break;
251 		default:
252 			moveDown();
253 		}
254 	}
255 	
256     /**
257      * Frogger dies
258      */
259 	public void die() {
260 		if (isAnimating)
261 			return;
262 		
263 		if (!cheating) {
264 		    AudioEfx.frogDie.play(0.2);
265 		    followObject = null;
266 		    isAlive = false;
267 		    currentFrame = 4;	// dead sprite   
268 		    game.GameLives--;
269 		    hw_hasMoved = true;
270 		}
271 		
272 		timeOfDeath = getTime();
273 		game.levelTimer = Main.DEFAULT_LEVEL_TIME;
274 	}
275 	
276 	/**
277 	 * Frogger reaches a goal
278 	 */
279 	public void reach(final Goal g) {
280 		if (g.isReached == false) {
281 			AudioEfx.frogGoal.play(0.4);
282 			game.GameScore += 100;
283 			game.GameScore += game.levelTimer;
284 			if (g.isBonus) {
285 				AudioEfx.bonus.play(0.2);
286 				game.GameLives++;
287 			}
288 			g.reached();
289 			resetFrog();
290 		}
291 		else {
292 			setPosition(g.getPosition());
293 		}
294 	}
295 	
296 	public void update(final long deltaMs) {
297 		if (game.GameLives <= 0)
298 			return;
299 		
300 		// if dead, stay dead for 2 seconds.
301 		if (!isAlive && timeOfDeath + 2000 < System.currentTimeMillis())
302 				resetFrog();
303 		
304 		updateAnimation();	
305 		updateFollow(deltaMs);
306 		setFrame(currentFrame);
307 		
308 		// Level timer stuff
309 		deltaTime += deltaMs;
310 		if (deltaTime > 1000) {
311 			deltaTime = 0;
312 			game.levelTimer--;
313 		}
314 		
315 		if (game.levelTimer <= 0)
316 			die();
317 	}
