26  package frogger;
27  
28  import java.util.List;
29  import java.util.Random;
30  import java.util.LinkedList;
31  
32  import jig.engine.ResourceFactory;
33  import jig.engine.audio.AudioState;
34  import jig.engine.audio.jsound.AudioClip;
35  import jig.engine.audio.jsound.AudioStream;
36  
37  /**
38   * Controls the audio effects
39   * 
40   * @author vitaliy
41   *
42   */
43  public class AudioEfx {
44  
45  	// These are referenced as to when to play the sound effects
46  	FroggerCollisionDetection fc;
47  	Frogger frog;
48  	
49  	public Random rand = new Random(System.currentTimeMillis());
50  	
51  	// Background music
52  	private AudioStream gameMusic;
53  	
54  	public static final String A_FX_PATH = Main.RSC_PATH + "ambient_fx/";
55  	
56  	public static AudioClip frogJump = ResourceFactory.getFactory().getAudioClip(
57  			Main.RSC_PATH + "jump.wav");
58  	
59  	public static AudioClip frogDie = ResourceFactory.getFactory().getAudioClip(
60  			Main.RSC_PATH + "frog_die.ogg");	
61  
62  	public static AudioClip frogGoal = ResourceFactory.getFactory().getAudioClip(
63  			Main.RSC_PATH + "goal.ogg");
64  	
65  	public static AudioClip levelGoal = ResourceFactory.getFactory().getAudioClip(
66  			Main.RSC_PATH + "level_goal.ogg");
67  
68  	public static AudioClip wind = ResourceFactory.getFactory().getAudioClip(
69  			Main.RSC_PATH + "wind.ogg");
70  	
71  	public static AudioClip heat = ResourceFactory.getFactory().getAudioClip(
72  			Main.RSC_PATH + "match.ogg");
73  	
74  	public static AudioClip bonus = ResourceFactory.getFactory().getAudioClip(
75  			Main.RSC_PATH + "bonus.ogg");	
76  	
77  	public static AudioClip siren = ResourceFactory.getFactory().getAudioClip(
78  			A_FX_PATH + "siren.ogg");
79  	
80  	// one effect is randomly picked from road_effects or water_effects every couple of seconds
81  	private List<AudioClip> road_effects = new LinkedList<AudioClip>();
82  	private List<AudioClip> water_effects = new LinkedList<AudioClip>();
83  	
84  	private int effectsDelay = 3000;
85  	private int deltaT = 0;
86  	
87  	/**
88  	 * In order to know when to play-back certain effects, we track the state of 
89  	 * collision detector and Frogger
90  	 * @param f
91  	 * @param frg
92  	 */
93  	public AudioEfx(FroggerCollisionDetection f, Frogger frg) {
94  		fc = f;
95  		frog = frg;
96  		
97  		road_effects.add(ResourceFactory.getFactory().getAudioClip(A_FX_PATH + "long-horn.ogg"));
98  	    road_effects.add(ResourceFactory.getFactory().getAudioClip(A_FX_PATH + "car-pass.ogg"));
99  		road_effects.add(ResourceFactory.getFactory().getAudioClip(A_FX_PATH + "siren.ogg"));
100 
101 		water_effects.add(ResourceFactory.getFactory().getAudioClip(A_FX_PATH + "water-splash.ogg"));
102 	    water_effects.add(ResourceFactory.getFactory().getAudioClip(A_FX_PATH + "splash.ogg"));
103 		water_effects.add(ResourceFactory.getFactory().getAudioClip(A_FX_PATH + "frog.ogg"));
104 
105 		gameMusic = new AudioStream(Main.RSC_PATH + "bg_music.ogg");
106 	}
107 	
108 	public void playGameMusic() {
109 	    gameMusic.loop(0.2, 0);
110 	}
111 	
112 	public void playCompleteLevel() {
113 		gameMusic.pause();
114 		levelGoal.play(2.0);
115 	}
116 	
117 	public void playRandomAmbientSound(final long deltaMs) {
118 		deltaT += deltaMs;
119 		
120 		if (deltaT > effectsDelay && fc.isOnRoad()) {
121 			deltaT = 0;
122 			road_effects.get(rand.nextInt(road_effects.size())).play(0.2);
123 		}
124 		
125 		if (deltaT > effectsDelay && fc.isInRiver()) {
126 			deltaT = 0;
127 			water_effects.get(rand.nextInt(road_effects.size())).play(0.2);
128 		}
129 	}
130 	
131 	public void update(final long deltaMs) {
132 		playRandomAmbientSound(deltaMs);
133 		
134 		if (frog.isAlive && (gameMusic.getState() == AudioState.PAUSED))
135 			gameMusic.resume();
136 		
137 		if (!frog.isAlive && (gameMusic.getState() == AudioState.PLAYING))
138 			gameMusic.pause();	
139 
140 	}
141 
142 }
